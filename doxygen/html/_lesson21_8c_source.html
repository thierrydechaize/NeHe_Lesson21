155 void TimerInit(void) // Initialize Our Timer (Get It Ready)
156 {
157  memset(&timer, 0, sizeof(timer)); // Clear Our Timer Structure
158 
159  // Check To See If A Performance Counter Is Available
160  // If One Is Available The Timer Frequency Will Be Updated
161  if (!QueryPerformanceFrequency((LARGE_INTEGER *) &timer.frequency))
162  {
163  // No Performace Counter Available
164  timer.performance_timer = FALSE; // Set Performance Timer To FALSE
165  timer.mm_timer_start = timeGetTime(); // Use timeGetTime() To Get Current Time
166  timer.resolution = 1.0f/1000.0f; // Set Our Timer Resolution To .001f
167  timer.frequency = 1000; // Set Our Timer Frequency To 1000
168  timer.mm_timer_elapsed = timer.mm_timer_start; // Set The Elapsed Time To The Current Time
169  }
170  else
171  {
172  // Performance Counter Is Available, Use It Instead Of The Multimedia Timer
173  // Get The Current Time And Store It In performance_timer_start
174  QueryPerformanceCounter((LARGE_INTEGER *) &timer.performance_timer_start);
175  timer.performance_timer = TRUE; // Set Performance Timer To TRUE
176  // Calculate The Timer Resolution Using The Timer Frequency
177  timer.resolution = (float) (((double)1.0f)/((double)timer.frequency));
178  // Set The Elapsed Time To The Current Time
179  timer.performance_timer_elapsed = timer.performance_timer_start;
180  }
181 }

183 float TimerGetTime() // Get Time In Milliseconds
184 {
185  __int64 time; // time Will Hold A 64 Bit Integer
186 
187  if (timer.performance_timer) // Are We Using The Performance Timer?
188  {
189  QueryPerformanceCounter((LARGE_INTEGER *) &time); // Grab The Current Performance Time
190  // Return The Current Time Minus The Start Time Multiplied By The Resolution And 1000 (To Get MS)
191  return ( (float) ( time - timer.performance_timer_start) * timer.resolution)*1000.0f;
192  }
193  else
194  {
195  // Return The Current Time Minus The Start Time Multiplied By The Resolution And 1000 (To Get MS)
196  return( (float) ( timeGetTime() - timer.mm_timer_start) * timer.resolution)*1000.0f;
197  }

216 bool NeHeLoadBitmap2(LPTSTR szFileName1, LPTSTR szFileName2, GLuint *texid) // Creates Texture From A Bitmap File

270 GLvoid BuildFont(GLvoid) // Build Our Font Display List
271 {
272  base=glGenLists(256); // Creating 256 Display Lists
273  glBindTexture(GL_TEXTURE_2D, texture[0]); // Select Our Font Texture
274  for (loop1=0; loop1<256; loop1++) // Loop Through All 256 Lists
275  {
276  float cx=(float)(loop1%16)/16.0f; // X Position Of Current Character
277  float cy=(float)(loop1/16)/16.0f; // Y Position Of Current Character
278 
279  glNewList(base+loop1,GL_COMPILE); // Start Building A List
280  glBegin(GL_QUADS); // Use A Quad For Each Character
281  glTexCoord2f(cx,1.0f-cy-0.0625f); // Texture Coord (Bottom Left)
282  glVertex2d(0,16); // Vertex Coord (Bottom Left)
283  glTexCoord2f(cx+0.0625f,1.0f-cy-0.0625f); // Texture Coord (Bottom Right)
284  glVertex2i(16,16); // Vertex Coord (Bottom Right)
285  glTexCoord2f(cx+0.0625f,1.0f-cy); // Texture Coord (Top Right)
286  glVertex2i(16,0); // Vertex Coord (Top Right)
287  glTexCoord2f(cx,1.0f-cy); // Texture Coord (Top Left)
288  glVertex2i(0,0); // Vertex Coord (Top Left)
289  glEnd(); // Done Building Our Quad (Character)
290  glTranslated(15,0,0); // Move To The Right Of The Character
291  glEndList(); // Done Building The Display List
292  } // Loop Until All 256 Are Built
